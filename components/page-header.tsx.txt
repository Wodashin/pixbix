import type { ReactNode } from "react"

interface PageHeaderProps {
  title: string
  description?: string
  children?: ReactNode
  gradient?: boolean
}

export function PageHeader({ title, description, children, gradient = true }: PageHeaderProps) {
  return (
    <div className="container mx-auto px-4 py-8">
      <div className="flex items-center justify-between mb-6">
        <div>
          <h1
            className={`text-4xl md:text-5xl font-bold mb-2 ${
              gradient ? "bg-gradient-to-r from-cyan-400 to-purple-600 bg-clip-text text-transparent" : "text-white"
            }`}
          >
            {title}
          </h1>
          {description && <p className="text-xl text-slate-400">{description}</p>}
        </div>
        {children && <div className="flex items-center gap-4">{children}</div>}
      </div>
    </div>
  )
}
